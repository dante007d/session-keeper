### A2. ANIMATED COUNTER — STAT CARDS

Numbers count up from 0 on mount. Uses spring physics.
The most emotionally satisfying animation on a dashboard.

```jsx
function AnimatedNumber({ value, duration = 1.4 }) {
  const [display, setDisplay] = React.useState(0)

  React.useEffect(() => {
    let start = 0
    const end = parseInt(value)
    if (start === end) return

    const increment = end / (duration * 60)  // 60fps
    const timer = setInterval(() => {
      start += increment
      if (start >= end) {
        setDisplay(end)
        clearInterval(timer)
      } else {
        setDisplay(Math.floor(start))
      }
    }, 1000 / 60)

    return () => clearInterval(timer)
  }, [value])

  // Pad to 2 digits like "07" — keeps layout stable
  return (
    <span className="font-mono font-bold text-5xl text-ink tabular-nums">
      {String(display).padStart(2, '0')}
    </span>
  )
}
```

### A4. MAGNETIC CURSOR

The cursor becomes a small amber dot with a larger ring that
follows with lag. On interactive elements, the ring expands
and blends into the element. Judges will notice immediately.

```jsx
function MagneticCursor() {
  const dotRef = React.useRef(null)
  const ringRef = React.useRef(null)
  const mouse = React.useRef({ x: 0, y: 0 })
  const ring = React.useRef({ x: 0, y: 0 })

  React.useEffect(() => {
    const onMove = (e) => {
      mouse.current = { x: e.clientX, y: e.clientY }
      // Dot follows instantly
      dotRef.current.style.transform =
        `translate(${e.clientX - 4}px, ${e.clientY - 4}px)`
    }

    // Ring follows with lerp lag
    const animate = () => {
      ring.current.x += (mouse.current.x - ring.current.x) * 0.12
      ring.current.y += (mouse.current.y - ring.current.y) * 0.12
      ringRef.current.style.transform =
        `translate(${ring.current.x - 16}px, ${ring.current.y - 16}px)`
      requestAnimationFrame(animate)
    }

    window.addEventListener('mousemove', onMove)
    animate()

    // Expand ring on interactive elements
    const interactives = document.querySelectorAll('button, a, input, [data-cursor]')
    interactives.forEach(el => {
      el.addEventListener('mouseenter', () => ringRef.current.classList.add('ring-expanded'))
      el.addEventListener('mouseleave', () => ringRef.current.classList.remove('ring-expanded'))
    })

    return () => window.removeEventListener('mousemove', onMove)
  }, [])

  return (
    <>
      <div ref={dotRef} className="cursor-dot" />
      <div ref={ringRef} className="cursor-ring" />
    </>
  )
}
```

### A7. CARD TILT ON HOVER (3D PERSPECTIVE)

Session cards in the grid tilt subtly on mouse position.
Feels premium, tactile, alive. Pure CSS transform.

```jsx
function TiltCard({ children, className }) {
  const cardRef = React.useRef(null)

  const handleMouseMove = (e) => {
    const rect = cardRef.current.getBoundingClientRect()
    const x = (e.clientX - rect.left) / rect.width   // 0 to 1
    const y = (e.clientY - rect.top) / rect.height    // 0 to 1
    const tiltX = (y - 0.5) * 8    // -4deg to +4deg
    const tiltY = (x - 0.5) * -8   // -4deg to +4deg

    cardRef.current.style.transform =
      `perspective(1000px) rotateX(${tiltX}deg) rotateY(${tiltY}deg) scale(1.01)`
  }

  const handleMouseLeave = () => {
    cardRef.current.style.transform =
      'perspective(1000px) rotateX(0deg) rotateY(0deg) scale(1)'
  }

  return (
    <div
      ref={cardRef}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      className={`transition-transform duration-200 ease-out ${className}`}
      style={{ transformStyle: 'preserve-3d' }}
    >
      {children}
    </div>
  )
}
```

```jsx
function FloatingInput({ label, required, ...props }) {
  const [focused, setFocused] = React.useState(false)
  const [hasValue, setHasValue] = React.useState(false)

  const isFloated = focused || hasValue

  return (
    <div className="relative">
      <motion.label
        animate={{
          y: isFloated ? -22 : 0,
          scale: isFloated ? 0.78 : 1,
          color: focused ? '#E8A020' : '#A09890',
        }}
        transition={{ duration: 0.2, ease: 'easeOut' }}
        className="absolute left-4 top-3.5 origin-left font-ui text-sm
                   pointer-events-none z-10"
      >
        {label}{required && <span className="text-gold ml-0.5">*</span>}
      </motion.label>

      <input
        {...props}
        onFocus={() => setFocused(true)}
        onBlur={(e) => { setFocused(false); setHasValue(!!e.target.value) }}
        onChange={(e) => { setHasValue(!!e.target.value); props.onChange?.(e) }}
        className="w-full pt-5 pb-2 px-4 rounded-xl bg-white border border-border
                   font-ui text-sm text-ink
                   focus:outline-none focus:border-gold focus:ring-2 focus:ring-gold/15
                   transition-all duration-150"
      />

      {/* Bottom progress line — fills on focus */}
      <motion.div
        className="absolute bottom-0 left-4 right-4 h-[2px] bg-gold rounded-full"
        initial={{ scaleX: 0 }}
        animate={{ scaleX: focused ? 1 : 0 }}
        transition={{ duration: 0.25 }}
        style={{ originX: 0 }}
      />
    </div>
  )
}
```

```jsx
function Toast({ message, type = 'success', onUndo, duration = 4000 }) {
  const [visible, setVisible] = React.useState(true)
  const [progress, setProgress] = React.useState(100)

  React.useEffect(() => {
    const start = Date.now()
    const tick = setInterval(() => {
      const elapsed = Date.now() - start
      const remaining = Math.max(0, 100 - (elapsed / duration) * 100)
      setProgress(remaining)
      if (remaining === 0) { setVisible(false); clearInterval(tick) }
    }, 16)
    return () => clearInterval(tick)
  }, [])

  return (
    <AnimatePresence>
      {visible && (
        <motion.div
          initial={{ opacity: 0, y: 16, scale: 0.96 }}
          animate={{ opacity: 1, y: 0, scale: 1 }}
          exit={{ opacity: 0, y: 8, scale: 0.96 }}
          transition={{ duration: 0.3, ease: [0.16, 1, 0.3, 1] }}
          className="fixed bottom-6 right-6 z-50 overflow-hidden
                     bg-ink text-cream rounded-2xl shadow-lg
                     min-w-[280px] max-w-[360px]"
        >
          <div className="flex items-center gap-3 px-5 py-4">
            <span className="text-base">
              {type === 'success' ? '✓' : type === 'info' ? 'ℹ' : '✕'}
            </span>
            <p className="font-ui text-sm flex-1">{message}</p>
            {onUndo && (
              <button
                onClick={onUndo}
                className="font-ui text-xs font-bold text-gold
                           hover:text-gold-bright transition-colors"
              >
                UNDO
              </button>
            )}
          </div>
          {/* Progress bar */}
          <div className="absolute bottom-0 left-0 h-[2px] bg-gold/40"
               style={{ width: `${progress}%`, transition: 'width 16ms linear' }} />
        </motion.div>
      )}
    </AnimatePresence>
  )
}
```

```jsx
function ArchiveSearch({ sessions, onFilter }) {
  const [query, setQuery] = React.useState('')

      s.title.toLowerCase().includes(q) ||
      s.host.toLowerCase().includes(q) ||
      s.tags?.some(t => t.includes(q)) ||
      s.summary?.toLowerCase().includes(q)

      <span className="absolute left-4 top-1/2 -translate-y-1/2 text-ink-muted text-sm">

        placeholder="Search sessions, hosts, tags..."

                   font-ui text-sm text-ink placeholder:text-ink-muted
                   focus:outline-none focus:border-gold focus:ring-2 focus:ring-gold/15

        <button
```

```jsx
function CountdownTimer({ targetDate }) {
  const [timeLeft, setTimeLeft] = React.useState(calcTimeLeft(targetDate))
```

```jsx
  doc.text(`Host: ${session.host}`, 20, 82)

  let y = 110
  session.attendance.forEach((member, i) => {

    doc.setTextColor('#000000')
    y += 10
  })

  doc.save(`${session.title.replace(/\s+/g, '_')}_report.pdf`)
```

### F9. 🌙 DARK MODE TOGGLE

One click switches to Midnight Amber dark theme.

```jsx
function ThemeToggle() {
  const [dark, setDark] = React.useState(
    () => localStorage.getItem('theme') === 'dark'
  )

  React.useEffect(() => {
    document.documentElement.setAttribute('data-theme', dark ? 'dark' : 'light')
    localStorage.setItem('theme', dark ? 'dark' : 'light')
  }, [dark])

  return (
    <motion.button
      whileTap={{ scale: 0.9 }}
      onClick={() => setDark(d => !d)}
      className="w-9 h-9 rounded-full flex items-center justify-center
                 bg-base border border-border text-ink-muted
                 hover:border-gold hover:text-gold transition-all duration-150"
    >
      <AnimatePresence mode="wait">
        <motion.span
          key={dark ? 'sun' : 'moon'}
          initial={{ rotate: -90, opacity: 0 }}
          animate={{ rotate: 0, opacity: 1 }}
          exit={{ rotate: 90, opacity: 0 }}
          transition={{ duration: 0.2 }}
          className="text-sm"
        >
          {dark ? '☀' : '☾'}
        </motion.span>
      </AnimatePresence>
    </motion.button>
  )
}
```

```jsx
function Leaderboard({ members, sessions }) {
  const ranked = members
    .map(m => ({
      ...m,
      attended: sessions.filter(s =>
        s.attendance.find(a => a.memberId === m.id && a.status === 'present')
      ).length,
      rate: sessions.length > 0
        ? Math.round((attended / sessions.length) * 100) : 0
    }))
    .sort((a, b) => b.rate - a.rate)

  return (
    <div className="space-y-3">
      {ranked.map((member, i) => (
        <motion.div
          key={member.id}
          initial={{ opacity: 0, x: -16 }}
          animate={{ opacity: 1, x: 0 }}
          transition={{ delay: i * 0.05, duration: 0.4, ease: [0.16,1,0.3,1] }}
          className="flex items-center gap-4"
        >
          {/* Rank */}
          <span className="font-mono font-bold text-sm w-6 text-right"
                style={{ color: i === 0 ? '#E8A020' : i === 1 ? '#A09070' : '#C4A882' }}>
            {String(i+1).padStart(2,'0')}
          </span>

          <MemberAvatar name={member.name} size="sm" />

          <div className="flex-1 min-w-0">
            <div className="flex justify-between mb-1">
              <span className="font-ui font-semibold text-sm text-ink truncate">
                {member.name}
              </span>
              <span className="font-mono font-bold text-sm text-gold ml-2">
                {member.rate}%
              </span>
            </div>
            {/* Animated bar */}
            <div className="w-full h-1.5 bg-base rounded-full overflow-hidden">

                className="h-full rounded-full"
```
